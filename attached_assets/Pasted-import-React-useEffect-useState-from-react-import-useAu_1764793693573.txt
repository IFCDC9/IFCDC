import React, { useEffect, useState } from 'react';
import { useAuth } from '../context/AuthContext';
import { getOverview, getIncidentsByProgram, getIncidentsTimeSeries } from '../api/reportsApi';
import { getHighRiskQueue, getMyQueue } from '../api/queueApi';

import StatsCards from '../components/StatsCards';
import IncidentsByProgramChart from '../components/IncidentsByProgramChart';
import IncidentsTimeSeriesChart from '../components/IncidentsTimeSeriesChart';
import QueuesPanel from '../components/QueuesPanel';

export default function DashboardPage() {
  const { token, user } = useAuth();

  const [overview, setOverview] = useState(null);
  const [byProgram, setByProgram] = useState(null);
  const [timeSeries, setTimeSeries] = useState(null);
  const [highRisk, setHighRisk] = useState([]);
  const [myQueue, setMyQueue] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState('');

  useEffect(() => {
    let isMounted = true;

    async function load() {
      setLoading(true);
      setError('');
      try {
        const [ov, prog, series, high, mine] = await Promise.all([
          getOverview(token),
          getIncidentsByProgram(token),
          getIncidentsTimeSeries(token, 30),
          getHighRiskQueue(token),
          getMyQueue(token),
        ]);

        if (!isMounted) return;
        setOverview(ov);
        setByProgram(prog);
        setTimeSeries(series);
        setHighRisk(high);
        setMyQueue(mine);
      } catch (err) {
        console.error(err);
        if (isMounted) setError('Error loading dashboard data.');
      } finally {
        if (isMounted) setLoading(false);
      }
    }

    if (token) {
      load();
    }

    return () => {
      isMounted = false;
    };
  }, [token]);

  return (
    <div className="dashboard-page">
      <header className="page-header">
        <div>
          <h1>IFCDC Command Console</h1>
          <p>Real-time view into incidents, intakes, and supervisory workload.</p>
        </div>
      </header>

      {error && <div className="error-banner">{error}</div>}

      {loading && !overview ? (
        <div>Loading dashboard…</div>
      ) : (
        <>
          <StatsCards overview={overview} />

          <div className="charts-grid">
            <IncidentsByProgramChart data={byProgram} />
            <IncidentsTimeSeriesChart data={timeSeries} />
          </div>

          <QueuesPanel
            highRisk={highRisk}
            myQueue={myQueue}
            currentUser={user}
          />
        </>
      )}
    </div>
  );
}
