    // Outreach tasks
    async function loadOutreachTasks() {
      const listEl = document.getElementById("outreach-list");
      listEl.textContent = "Loading outreach tasks...";
      try {
        const res = await fetch("/api/outreach-tasks?status=OPEN", {
          headers: getAuthHeaders(),
        });
        const data = await res.json();
        if (!res.ok) {
          listEl.textContent = data.error || "Failed to load outreach tasks.";
          return;
        }
        listEl.innerHTML = "";
        if (!data.length) {
          listEl.textContent = "No open outreach tasks.";
          return;
        }
        data.forEach((t) => {
          const div = document.createElement("div");
          div.className = "admin-list-item";
          div.innerHTML =
            "<strong>" +
            (t.clientName || "Unknown caller") +
            "</strong><br/>" +
            "<span style='opacity:0.8;'>Phone: " +
            t.phone +
            "</span><br/>" +
            "<span style='opacity:0.7; font-size:0.78rem;'>Reason: " +
            t.reason +
            " | Created: " +
            t.createdAt +
            "</span>";

          const btn = document.createElement("button");
          btn.textContent = "Mark Completed";
          btn.className = "ifcdc-btn ifcdc-btn-primary";
          btn.style.marginTop = "0.25rem";
          btn.onclick = async (ev) => {
            ev.stopPropagation();
            try {
              const res2 = await fetch(
                "/api/outreach-tasks/" + t.id + "/complete",
                {
                  method: "POST",
                  headers: getAuthHeaders(),
                  body: JSON.stringify({}),
                }
              );
              const d2 = await res2.json();
              if (!res2.ok) {
                alert(d2.error || "Failed to complete task.");
                return;
              }
              loadOutreachTasks();
            } catch (err) {
              alert("Error completing task.");
            }
          };

          div.appendChild(document.createElement("br"));
          div.appendChild(btn);
          listEl.appendChild(div);
        });
      } catch (err) {
        listEl.textContent = "Error loading outreach tasks.";
      }
    }

    document.getElementById("outreach-refresh").onclick = loadOutreachTasks;
