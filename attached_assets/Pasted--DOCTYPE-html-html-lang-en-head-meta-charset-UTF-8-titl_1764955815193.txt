  <script>
    document.getElementById('login-form').addEventListener('submit', async (e) => {
      e.preventDefault();

      const email = document.getElementById('email').value.trim();
      const password = document.getElementById('password').value.trim();
      const msgBox = document.getElementById('login-message');
      msgBox.innerHTML = '';

      try {
        const res = await fetch('/api/auth/login', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ email, password })
        });

        const data = await res.json();

        if (!res.ok) {
          msgBox.innerHTML = `<p class="auth-error">${data.error || 'Login failed'}</p>`;
          return;
        }

        msgBox.innerHTML = `<p class="auth-success">Login successful. Redirecting…</p>`;

        // Redirect by role
        if (data.role === 'admin') {
          window.location.href = '/admin.html';
        } else if (data.role === 'barber') {
          window.location.href = '/barber.html';
        } else if (data.role === 'radio') {
          window.location.href = '/radio.html';
        } else {
          window.location.href = '/client.html';
        }
      } catch (err) {
        msgBox.innerHTML = `<p class="auth-error">Network error. Try again.</p>`;
      }
    });
  </script>
