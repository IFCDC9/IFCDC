  <script>
    document.getElementById('register-form').addEventListener('submit', async (e) => {
      e.preventDefault();

      const name = document.getElementById('name').value.trim();
      const email = document.getElementById('email').value.trim();
      const password = document.getElementById('password').value.trim();
      const role = document.getElementById('role').value;
      const msgBox = document.getElementById('register-message');
      msgBox.innerHTML = '';

      try {
        const res = await fetch('/api/auth/register', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ name, email, password, role })
        });

        const data = await res.json();

        if (!res.ok) {
          msgBox.innerHTML = `<p class="auth-error">${data.error || 'Registration failed'}</p>`;
          return;
        }

        msgBox.innerHTML = `<p class="auth-success">Account created. Redirecting to login…</p>`;
        setTimeout(() => {
          window.location.href = '/login.html';
        }, 1000);
      } catch (err) {
        msgBox.innerHTML = `<p class="auth-error">Network error. Try again.</p>`;
      }
    });
  </script>
